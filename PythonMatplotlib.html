# Make Basic Graph.
x = [0,1,2,3,4,5]#X-Axis-Values.
y = [0,2,4,6,8,10]#Y-Axis-Values.
plt.plot(x,y,
         label='2x',
         color='red',
         linewidth=2,
         marker='.',
        markersize=10,
         markeredgecolor='blue',
        linestyle='--')#(X,Y).
#>>Graph-Title<<.
plt.title('Our First Graph',
          fontdict={'fontname':'Comic Sans MS',
         'fontsize':20,'color':'red'})
#>>Label-Text<<.
plt.xlabel('X Axis',fontdict={'fontname':'Comic Sans MS'})
plt.ylabel('Y Axis',fontdict={'fontname':'Comic Sans MS'})
#-------------------------
plt.xticks([0,1,2,3,4,5])
plt.yticks([0,2,4,6,8,10])
#-------------------------
plt.legend()
plt.show()
